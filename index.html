Quick install on IBM i

1 — Download and enable the repository

curl -o /QOpenSys/etc/yum/repos.d/ibmi-rpm-packages.repo \
     https://ludovic-gasc.github.io/ibmi-rpm-packages/ibmi-rpm-packages.repo

2 — Import the GPG key

curl -o /tmp/RPM-GPG-KEY-ibmi \
     https://ludovic-gasc.github.io/ibmi-rpm-packages/RPM-GPG-KEY-ibmi
rpm --import /tmp/RPM-GPG-KEY-ibmi

3 — Install a package

yum makecache
yum install qpdf
